    <script>
        function toggleTopic(button) {
            const card = button.closest('.topic-card');
            const details = card.querySelector('.topic-details');
            if (details.style.display === 'block') {
                details.style.display = 'none';
                button.innerText = 'Развернуть';
                button.style.background = '#27ae60'; // Обычный зелёный
            } else {
                details.style.display = 'block';
                button.innerText = 'Свернуть';
                button.style.background = '#218c4e'; // Благородный тёмно-зелёный вместо красного
            }
        }
